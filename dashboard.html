# Creating a zip file containing the updated dashboard HTML files for different user types from zipfile import ZipFile import os # Define the folder and file names folder_path = "/mnt/data/studenthub_dashboard" os.makedirs(folder_path, exist_ok=True) # Create HTML content for different dashboard views dashboard_buyer = """ Buyer Dashboard

Welcome Buyer!

You can browse, search and purchase products.

""" dashboard_seller = """ Seller Dashboard

Welcome Seller!

You can upload, edit and manage your product listings.

""" dashboard_entrepreneur = """ Entrepreneur Dashboard

Welcome Entrepreneur!

You can submit startup ideas, track investments, and find collaborators.

""" # Save the files file_paths = [] for filename, content in [("dashboard_buyer.html", dashboard_buyer), ("dashboard_seller.html", dashboard_seller), ("dashboard_entrepreneur.html", dashboard_entrepreneur)]: file_path = os.path.join(folder_path, filename) with open(file_path, "w") as f: f.write(content) file_paths.append(file_path) # Create a zip archive of the dashboard files zip_path = "/mnt/data/dashboards.zip" with ZipFile(zip_path, 'w') as zipf: for file in file_paths: zipf.write(file, os.path.basename(file)) zip_path
